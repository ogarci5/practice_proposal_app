def create
  @user = User.find(params[:proposal][:user_id])
  @proposal = @user.proposals.new(name: params[:proposal][:name], 
  description: params[:proposal][:description], from: current_user.name, response: "")
  if @proposal.save
    redirect_to proposals_path
  end
end
